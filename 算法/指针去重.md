const uniqueString = function (str) {
    var maxRepeatChar   = '',
    maxRepeatCharCount  = 0,
                    j   = 1,
                    i   = 0;
    while (i <= str.length - 1) {
        if (str[i] !== str[j]) {
            // 如果当前字符的重复次数超过了之前记录的最大重复次数，那么最大重复次数倍重新赋值
            if (j - i > maxRepeatCharCount) {
                maxRepeatCharCount = j - i;
                maxRepeatChar = str[i]
            }
            i = j;
        } 
       j++;
    }
    return {
        maxRepeatChar,
        maxRepeatCharCount
    }
}
